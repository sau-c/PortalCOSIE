    BorderColor #5D6D7E

    FontSize 13

    AttributeFontColor #333333
    AttributeFontSize 11
}

' Estilos semánticos
skinparam state {
    BackgroundColor<<Inicial>> #EBF5FB
    BorderColor<<Inicial>> #2E86C1
    
    BackgroundColor<<Exito>> #D5F5E3
    BorderColor<<Exito>> #196F3D
    
    BackgroundColor<<Warning>> #FCF3CF
    BorderColor<<Warning>> #F1C40F
    
    BackgroundColor<<Error>> #FADBD8
    BorderColor<<Error>> #943126
}
hide empty description

' --- Definición de Estados (Basado en tu C#) ---
state "En Revisión" as Revision <<Inicial>> : (ID: 1) Estado inicial.\nEl documento espera validación.

state "Validado" as Validado <<Exito>> : (ID: 2) Cumple requisitos.\nFin del proceso para este doc.

state "Con Errores" as ConErrores <<Warning>> : (ID: 3) Problemas corregibles:\nFirmas, sellos o fechas.

state "Documento Incorrecto" as Incorrecto <<Error>> : (ID: 4) No es el solicitado.\nArchivo erróneo.

' --- Flujo y Transiciones ---

[*] --> Revision : Carga inicial \n(Alumno)

' Caminos de decisión
Revision -down-> Validado : Todo correcto
Revision -right-> ConErrores : Detalles \nmenores
Revision -left-> Incorrecto : Archivo no \ncorresponde

' Ciclos de corrección (Vuelven a revisión)
ConErrores --> Revision : Corrige y \nvuelve a subir
Incorrecto --> Revision : Sube el archivo \ncorrecto

' Final
Validado --> [*]

@enduml
